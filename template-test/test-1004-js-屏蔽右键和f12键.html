屏蔽右键和f12键

屏蔽右键

测试：右键试试 右键已经被屏蔽了！

测试：f12试试 f12已经被屏蔽了！

    	var txt1 = document.getElementById("txt1");
        var txt2 = document.getElementById("txt2");
        //屏蔽右键
        document.oncontextmenu = function(){
            txt1.style.display = 'inline-block';
            return false;
        };

        //屏蔽f12
        document.onkeydown = function(ev){
            if(ev.keyCode === 123){ 
                txt2.style.display = 'inline-block';
                return false;
            }
        };
