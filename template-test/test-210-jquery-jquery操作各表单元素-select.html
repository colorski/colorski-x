1.获取下拉框选中项的值、文本、索引

        $("#id").find('option:selected').text();

		console.log($("#test1 option:selected").val());
		console.log($("#test1").find('option:selected').text());

		$("#test1").change(function(){

				'值='+$(this).val() + 
				'文本='+$(this).find('option:selected').text() + 
				'索引='+$(this).get(0).selectedIndex);
