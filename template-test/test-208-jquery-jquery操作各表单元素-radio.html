jquery操作各表单元素-radio

1.获取单选按钮的值
    	var v_radio = $(":radio[name='iptName']:checked").val();

实例代码：

    	var btn = $('#btn1');
    	//默认选中项的值
    	var v_radio = $(":radio[name='iptName']:checked").val();

    	$(":radio[name='iptName']").change(function() {
    		//改变选择后选中项的值
    		v_radio = $(this).val(); 

    		//根据选中项做一些事
    		if(v_radio == 3){
    			alert('我被选中了！我的值是'+v_radio);
    		}
    	});
    	
    	btn.click(function(){
    		alert(v_radio);
    	});
