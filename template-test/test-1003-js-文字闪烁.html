原生js写的文字闪烁

测试：你看我是在闪吗^_^？

    	//闪烁定义
    	function shine(obj){
    		var i=0;
    		var timer=null;
    		clearInterval(timer);
    		timer=setInterval(function(){
    			i++;
    			if(i%2){
    				obj.style='color:#C6F;';
    			}else{
    				obj.style='color:#F00;';
    			}
    			if(i==6){
    				clearInterval(timer);
    				i=0;
    			}
    		},60);
    	};
    	var oTxt=document.getElementById("shine_text");
    	var oBtn=document.getElementById("btn1");

    	//闪烁执行
    	oBtn.onclick=function(){
    		shine(oTxt);
    	};
